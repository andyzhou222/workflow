#!/bin/bash
set -e   # 遇到错误立即停止

echo "========== START DEPLOY =========="

# 输出环境变量调试
echo "PORT=$PORT"
echo "DATABASE_URL=$DATABASE_URL"

# 切换到后端目录
cd backend

echo "Installing dependencies..."
pip install -r requirements.txt

echo "Starting FastAPI server..."
uvicorn main:app --host 0.0.0.0 --port $PORT

echo "========== END DEPLOY =========="
